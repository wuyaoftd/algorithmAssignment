[350.两个数组的交集 II](https://leetcode-cn.com/problems/intersection-of-two-arrays-ii/)
```js
/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number[]}
 */
var intersect = function(nums1, nums2) {
    let result =[]
    nums1.forEach(item=>{
        let index =nums2.indexOf(item)
        if(index>-1){
            result.push(item)
            nums2.splice(index,1)
        }
    })
    return result
};
```
